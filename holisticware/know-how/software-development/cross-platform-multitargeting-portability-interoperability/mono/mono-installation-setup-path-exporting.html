Mono Setup Path Exporting

#!/bin/bash -e

# !!!!!!!!!!!!! execute this script with ROOT privileges !!!!!!!!!!!!!
# exporting PATH for MONO; system wide; /etc/profile
# tested on UBUNTU SERVER 11.4
# before executing change permissions
# chmod 755 ./install_mono-2.10.2_paths.sh

PATHS="
export PATH=/opt/mono-2.10/bin:$PATH
export PKG_CONFIG_PATH=/opt/mono-2.10/lib/pkgconfig:$PKG_CONFIG_PATH
export LD_LIBRARY_PATH=/opt/mono-2.10/lib:$LD_LIBRARY_PATH
"

echo "$PATHS" >> /etc/profile

# You can also use echo $PATHS but then bash will interpret commands and we dont 
# want that we want import our config files into /etc/profile in exact way as
