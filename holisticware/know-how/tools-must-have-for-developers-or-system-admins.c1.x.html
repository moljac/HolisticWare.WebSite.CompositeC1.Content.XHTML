  • batch script to replace notepad (MS) with notepad++ 
    echo off

:: elevate is exe for vista/windows 7  UAC!
set ELEVATE=

%ELEVATE% xcopy npp.5.7.bin\unicode\notepad++.exe ^
	c:\windows\system32\dllcache\
%ELEVATE% xcopy npp.5.7.bin\unicode\notepad++.exe ^
	c:\windows\system32\
%ELEVATE% xcopy npp.5.7.bin\unicode\notepad++.exe ^
	c:\windows\
	
%ELEVATE% copy /-y npp.5.7.bin\unicode\notepad++.exe ^
	C:\WINDOWS\System32\dllcache\notepad.exe
%ELEVATE% copy /-y npp.5.7.bin\unicode\notepad++.exe ^
	C:\WINDOWS\ServicePackFiles\i386\notepad.exe
%ELEVATE% copy /-y npp.5.7.bin\unicode\notepad++.exe ^
	C:\WINDOWS\System32\notepad.exe
%ELEVATE% copy /-y npp.5.7.bin\unicode\notepad++.exe ^
	C:\WINDOWS\notepad.exe

echo ==========================================================================
echo Check for errors!
pause
  • Useful registry entries (remember notepad is now notepad++:
    • Add "Open with Notepad" to the Context Menu for All Files

[HKEY_CLASSES_ROOT\*\shell\Open with Notepad]

[HKEY_CLASSES_ROOT\*\shell\Open with Notepad\command]
