# Cisco 5XXG
# phone example provisioning files included

1. set up tftpd with detailed logging

	vi /etc/xinetd.d/tftp
	
	find this line:
	
	server_args             = -s /tftpboot
	
	and change to this:
	
	server_args             = -s /tftpboot -v -v -v
	
	save and quit.
	
2. restart xinetd

	/etc/init.d/xinetd restart
	
3. copy spa5XXG.cfg and spa5XXG_MAC_[mac address of the phone, all lower case, example: ccef485ce288].cfg

4. remove XMLDefault.xml.cnf or SEP_[mac address of the phone].cnf or XMLDefault5XXG.xml.cnf or the phone will use sccp instead of SIP.

5. add phone MAC address to DHCP if using any.

	if you expirience any problems with tftp or dhcp try add this to dhcpd.conf
	
	# cisco
	option option-150 code 150 = ip-address; 
	option voip-tftp-servers code 150 = ip-address;
	
	option option-66 code 66 = ip-address; 
	option voip-tftp-servers code 66 = ip-address;
	
	option option-3 code 3 = ip-address; 
	option voip-tftp-servers code 3 = ip-address;
	
	option voip-tftp-servers [ip address of your PBX (Elastix)];

6. run tail -f /var/log/messages

7. plug in the phone and watch loggs
